79  public void addPlayer(IArenaPlayer player) {
80  PreCon.notNull(player);
81 
82  _cachedNextGroup.reset();
83  _cachedReadyGroup = null;
84  _players.add(player);
85 
86  if (player.getPlayerGroup() != null) {
87 
88  // add existing player group
89  _groups.add(player.getPlayerGroup());
90  }
91  else {
92 
93  // Add new player to an existing group if available
94  for (IArenaPlayerGroup group : _groups) {
95  if (group.size() < _arena.getSettings().getMaxPlayers()) {
96  group.addPlayer(player);
97  return;
98  }
99  }
100 
101  // add new player to new group
102  IArenaPlayerGroup group = new ArenaPlayerGroup();
103  group.addPlayer(player);
104  _groups.add(group);
105  }
106  }

114  public void removePlayer(IArenaPlayer player, RemoveFromContextReason reason) {
115  PreCon.notNull(player);
116 
117  _cachedNextGroup.reset();
118  _cachedReadyGroup = null;
119 
120  _players.remove(player);
121 
122  IArenaPlayerGroup group = player.getPlayerGroup();
123  if (group == null)
124  throw new NullPointerException(
125  "Player was expected to be in a player group but was not.");
126 
127 
128  if (reason != RemoveFromContextReason.FORWARDING &&
129  reason != RemoveFromContextReason.CONTEXT_CHANGE) {
130  group.removePlayer(player);
131  }
132 
133  IArenaPlayerCollection groupPlayers = group.getPlayers();
134 
135  // check to see if any other players are in the group
136  for (IArenaPlayer groupPlayer : groupPlayers) {
137  if (_players.contains(groupPlayer)) {
138  return; // finished
139  }
140  }
141 
142  // no players in group are in the collection so remove group
143  _groups.remove(group);
144  }

270  @Nullable

272 
273  if (_cachedNextGroup.keyEquals(minSize))
274  return _cachedNextGroup.getValue();
275 
276  for (IArenaPlayerGroup group : _groups) {
277 
278  // find a group with players who meet the minimum group size
279  IArenaPlayerCollection players = group.filterPlayers(_players);
280  if (players.size() >= minSize) {
281  _cachedNextGroup.set(minSize, players);
282  return new ArenaPlayerArrayList(players, true);
283  }
284  }
285 
286  // return a ready group regardless of size
287  // to ensure groups that get split into smaller
288  // groups across multiple arenas can still play.
289  IArenaPlayerCollection result = getReadyGroup();
