135  /*
136  * Attempt to start a game automatically.
137  */
138  private boolean tryAutoStart() {
139 
140  IGameContext gameManager = getArena().getGame();
141 
142  // make sure game isn't already running
143  if (gameManager.isRunning())
144  return false;
145 
146  // make sure arena isn't busy
147  if (getArena().isBusy())
148  return false;
149 
150  // make sure arena has auto start enabled
151  // and count down isn't already running.
152  if (getSettings().hasAutoStart()) {
153 
154  // get the next group
155  IArenaPlayerCollection nextGroup = getNextGroup();
156 
157  // group not found
158  if (nextGroup == null || nextGroup.isEmpty())
159  return false;
160 
161  // make sure group meets the minimum auto start player size setting
162  if (nextGroup.size() < getSettings().getMinAutoStartPlayers())
163  return false;
164 
165  // start
166  return getArena().getGame().start(ArenaStartReason.AUTO);
167  }
168 
169  return false;
170  }

172  /*
173  * Attempt to start a game for ready players.
174  */
175  private boolean tryReadyStart() {
176 
177  // make sure game isn't running
178  if (getArena().getGame().isRunning())
179  return false;
180 
181  // make sure arena isn't busy
182  if (getArena().isBusy())
183  return false;
184 
185  // check to see if there is a group that is ready
186  IArenaPlayerCollection ready = getArena().getLobby().getReadyGroup();
187  if (ready == null || ready.isEmpty())
188  return false;
189 
190  // make sure the size of the group meets the min
191  // players requirement
192  return ready.size() >= getArena().getSettings().getMinPlayers() &&
193  // start
194  getArena().getGame().start(ArenaStartReason.PLAYERS_READY);
195 
196  }
