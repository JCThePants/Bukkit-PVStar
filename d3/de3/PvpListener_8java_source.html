56

60  IArenaPlayer arenaPlayer = ArenaPlayer.get(event.getPlayer());
61  IArena arena = arenaPlayer.getArena();
62  if (arena == null)
63  return;
64 
65  IContextSettings settings = arenaPlayer.getContextSettings();
66  if (settings == null)
67  return;
68 
69  if (settings.isToolsDamageable())
70  return;
71 
72  ItemStack inHand = event.getPlayer().getItemInHand();
73  if (inHand == null || inHand.getType() == Material.AIR)
74  return;
75 
76  Material material = inHand.getType();

146  Handle PVP
147  */
148  @EventHandler(priority= EventPriority.NORMAL, ignoreCancelled = true)
149  private void onPVP(EntityDamageByEntityEvent event) {
150 
151  Entity entity = event.getEntity();
152 
153  if (!(entity instanceof Player))
154  return;
155 
156  Player p = (Player)entity;
157 
158  IArenaPlayer player = ArenaPlayer.get(p);
159 
160  IArena arena = player.getArena();
161  if (arena == null)
162  return;
163 
164  // no damage when game is over
165  if (arena.getGame().isGameOver()) {
166  event.setDamage(0.0);
167  event.setCancelled(true);
168  return;
169  }
170 
171 
172  // get settings
173  IContextSettings settings = player.getContextSettings();
174  if (settings == null)
175  return;
176 
177  // prevent pvp
178  if (!settings.isPvpEnabled() || !settings.isTeamPvpEnabled()) {
179 
180  Entity damagerEntity = event.getDamager();
181  Player damager = null;
182 
183  // get damager
184  if (damagerEntity instanceof Projectile) {
185  ProjectileSource source = ((Projectile) damagerEntity).getShooter();
186 
187  if (source instanceof Player) {
188  damager = (Player) source;
189  }
190  } else if (damagerEntity instanceof Player) {
191  damager = (Player) damagerEntity;
192  }
193 
194  // handle player on player damage
195  if (damager != null) {
196 
197  // check for pvp
198  if (!settings.isPvpEnabled()) {
199  event.setDamage(0.0);
200  event.setCancelled(true);
201  }
202  // check for team pvp
203  else //noinspection ConstantConditions
204  if (!settings.isTeamPvpEnabled()) { // always true, statement is for readability
205  IArenaPlayer damagerPlayer = ArenaPlayer.get(p);
206 
207  // prevent team pvp
208  if (damagerPlayer.getTeam() == player.getTeam() &&
209  damagerPlayer.getTeam() != ArenaTeam.NONE ||
210  player.getTeam() != ArenaTeam.NONE) {
211 
212  event.setDamage(0.0D);
213  event.setCancelled(true);
214  }
215 
216  }
217  }
218  }
