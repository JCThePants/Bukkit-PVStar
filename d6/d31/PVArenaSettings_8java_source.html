101  @Override
102  public void setTransientEnabled(boolean isEnabled) {
103 
104  // Note: The setting is not checked to see if it is the same
105  // because the method may be called simply to run the events.
106  // (i.e. The arenas world is loaded or unloaded and events need)
107 
108  // prevent enabling if world is not loaded
109  if (_arena.getRegion().isDefined() &&
110  !_arena.getRegion().isWorldLoaded()) {
111 
112  if (isEnabled)
113  return;
114  }
115 
116  _isEnabled = isEnabled;
117  if (isEnabled) {
118 
119  if (_arena.getEventManager().call(this, new ArenaPreEnableEvent(_arena)).isCancelled())
120  return;
121 
122  _isEnabled = true;
123 
124  _arena.getExtensions().enable();
125  _arena.getEventManager().call(this, new ArenaEnabledEvent(_arena));
126  }
127  else {
128  _isEnabled = false;
129 
130  _arena.getExtensions().disable();
131  _arena.getEventManager().call(this, new ArenaDisabledEvent(_arena));
132  }
133  }
