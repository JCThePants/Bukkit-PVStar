38 @CommandInfo(
39  parent="lobby",

43

48 public class TeamPvpSubCommand extends AbstractPVCommand implements IExecutableCommand {
49 
50  @Localizable static final String _TEAM_PVP_ENABLED =
51  "Arena '{0: arena name}' lobby Team PVP is enabled.";
52 
53  @Localizable static final String _TEAM_PVP_DISABLED =
54  "Arena '{0: arena name}' lobby Team PVP is {RED}disabled.";
55 
56  @Localizable static final String _TEAM_PVP_CHANGE_ENABLED =
57  "Arena '{0: arena name}' lobby Team PVP changed to enabled.";
58 
59  @Localizable static final String _TEAM_PVP_CHANGE_DISABLED =
60  "Arena '{0: arena name}' lobby Team PVP changed to {RED}disabled.";
61 
62  @Override
63  public void execute(CommandSender sender, ICommandArguments args) throws CommandException {
64 
65  IArena arena = getSelectedArena(sender, ArenaReturned.getInfoToggled(args, "on|off|info"));
66  if (arena == null)
67  return; // finish
68 
69  if (args.getString("on|off|info").equals("info")) {
70 
71  boolean isEnabled = arena.getLobby().getSettings().isTeamPvpEnabled();
72 
73  if (isEnabled) {
74  tell(sender, Lang.get(_TEAM_PVP_ENABLED, arena.getName()));
75  }
76  else {
77  tell(sender, Lang.get(_TEAM_PVP_DISABLED, arena.getName()));
78  }
79  }
80  else {
81 
82  boolean isEnabled = args.getBoolean("on|off|info");
83 
84  arena.getLobby().getSettings().setTeamPvpEnabled(isEnabled);
85 
86  if (isEnabled) {
87  tellSuccess(sender, Lang.get(_TEAM_PVP_CHANGE_ENABLED, arena.getName()));
88  }
89  else {
90  tellSuccess(sender, Lang.get(_TEAM_PVP_CHANGE_DISABLED, arena.getName()));
91  }
92  }
93  }
94 }
