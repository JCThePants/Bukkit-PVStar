97  public void enable(final Runnable onModulesEnabled) {
98  PreCon.notNull(onModulesEnabled);
99  PreCon.isValid(!_isModulesLoaded, "Modules can only be enabled once.");
100 
101  DependencyRunner<UnloadedModuleContainer> moduleEnabler =
102  new DependencyRunner<UnloadedModuleContainer>(PVStarAPI.getPlugin());
103 
104  moduleEnabler.addAll(_unloadedModules.values());
105  moduleEnabler.onFinish(new IFinishHandler<UnloadedModuleContainer>() {
106  @Override
107  public void onFinish(List<UnloadedModuleContainer> notRun) {
108 
109  // remove unloaded modules
110  for (UnloadedModuleContainer container : notRun) {
111  removeModule(container.getSearchName());
112 
113  Msg.warning("[{0}] Failed to load module because required dependencies are missing:", container.getName());
114 
115  List<String> bukkitDepends = container.getMissingBukkitDepends();
116  for (String depend : bukkitDepends) {
117  Msg.warning("[{0}] Missing Bukkit Plugin: {1}", container.getName(), depend);
118  }
119 
120  List<String> moduleDepends = container.getMissingModuleDepends();
121  for (String depend : moduleDepends) {
122  Msg.warning("[{0}] Missing PV-Star Module: {1}", container.getName(), depend);
123  }
124  }
125 
126  _unloadedModules.clear();
127 
128  // Enable Modules
129  for (PVStarModule module : getModules()) {
130  REGISTRATION.enable(module);
131  Msg.info("[{0}] Module enabled.", module.getName());
132  }
133 
134  _isModulesLoaded = true;
135 
136  onModulesEnabled.run();
137 
138  for (PVStarModule module : getModules()) {
139  REGISTRATION.arenasEnabled(module);
140  }
141 
142  HandlerList.unregisterAll(_kickPlayersListener);
143  }
144  });
145 
146  moduleEnabler.start();
147  }
