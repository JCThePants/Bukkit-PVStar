140  event.getReason() != AddToContextReason.CONTEXT_CHANGE) {

142  ArenaTeam team = nextTeam();

176  private void onTeamChange(PlayerTeamChangedEvent event) {
177 
178  if (event.getReason() == TeamChangeReason.JOIN_ARENA)
179  return;
180 
181  ArenaTeam previous = event.getPreviousTeam();
182  ArenaTeam current = event.getTeam();
183 
184  if (previous != ArenaTeam.NONE &&
185  previous != null) {
186 
187  _currentTeams.subtract(previous);
188  }
189 
190  if (current != ArenaTeam.NONE &&
191  current != null) {
192 
193  _currentTeams.add(current);
194  }
195  }
