126 
127  List<IArena> results = PVStarAPI.getArenaManager().getArena(arenaName, NameMatchMode.CASE_INSENSITIVE);
128 
129  if (results.size() == 0) {
130  results = PVStarAPI.getArenaManager().getArena(arenaName, NameMatchMode.BEGINS_WITH);
131  }
132 
133  if (results.size() == 0) {
134  Msg.tellError(sender, Lang.get(_ARENA_NOT_EXISTS, arenaName));
135  }
136  else if (results.size() == 1) {
137  return results.get(0);
138  }
139  else {
140  Msg.tellError(sender, Lang.get(_MULTIPLE_ARENAS, "{YELLOW}" + arenaName));
141  Msg.tell(sender, TextUtils.concat(results, ", "));
