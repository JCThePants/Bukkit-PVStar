374  public boolean remove(IArenaPlayer player, PlayerLeaveArenaReason reason) {
375  PreCon.isValid(player instanceof ArenaPlayer);
376  PreCon.notNull(reason);
377  PreCon.isValid(reason != PlayerLeaveArenaReason.FORWARDING);
378 
379  // make sure the player is in the game if they are being removed as a loss
380  if (reason == PlayerLeaveArenaReason.LOSE && player.getContext() != ArenaContext.GAME)
381  return false;
382 
383  AbstractContextManager manager = (AbstractContextManager)player.getContextManager();
384  if (manager != null) {
385 
386  Result<Location> restoreLocation = manager.removePlayer(player, reason.getContextEquivalent());
387  if (restoreLocation.hasResult()) {
388 
389  PlayerLeaveArenaEvent leaveEvent = new PlayerLeaveArenaEvent(
390  this, player, manager, reason, restoreLocation.getResult());
391 
392  getEventManager().call(this, leaveEvent);
393 
394  ((ArenaPlayer)player).clearArena();
395 
396  if (leaveEvent.isRestoring() &&
397  leaveEvent.getRestoreLocation() != null) {
398 
399  if (player.getPlayer().isDead()) {
400  ((ArenaPlayer)player).setDeathRespawnLocation(leaveEvent.getRestoreLocation());
401  }
402  else {
403  player.getPlayer().teleport(leaveEvent.getRestoreLocation());
404  }
405  }
406 
407  return true;
408  }
409  }

411  return false;
