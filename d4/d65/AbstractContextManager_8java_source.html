84  public final boolean addPlayer(IArenaPlayer player, AddToContextReason reason) {
85  PreCon.isValid(player instanceof ArenaPlayer);
86  PreCon.notNull(reason);
87 
88  if (_players.hasPlayer(player))
89  return false;
90 
91  if (!getArena().getSettings().isEnabled())
92  return false;
93 
94  if (reason != AddToContextReason.CONTEXT_CHANGE) {
95 
96  if (_arena.getEventManager().call(this,
97  new PlayerPreAddToContextEvent(_arena, player, this, reason)).isCancelled())
98  return false;
99  }
100 
101  _players.addPlayer(player);
102 
103  ((ArenaPlayer)player).setCurrentArena(_arena);
104 
105  Location spawnPoint = onPrePlayerAdd(player, reason);
106 
107  PlayerAddToContextEvent contextEvent = new PlayerAddToContextEvent(
108  _arena, player, this, getContext(), reason, spawnPoint, null);
109  _arena.getEventManager().call(this, contextEvent);
110 
111  contextEvent = onPlayerAdded(player, reason, contextEvent);
112 
113  TeleportMode teleportMode = getSettings().getTeleportMode();
114  Location location = contextEvent.getSpawnLocation();
115  // teleport player to spawn location from event
116  if (contextEvent.getSpawnLocation() != null
117  && Teleporter.teleport(player.getPlayer(), location, teleportMode).isSuccess()) {
118 
119  PlayerArenaSpawnedEvent spawnEvent = new PlayerArenaSpawnedEvent(
120  _arena, player, this, contextEvent.getSpawnLocation());
121  _arena.getEventManager().call(this, spawnEvent);
122 
123  // reserve spawnpoint
124  if (contextEvent.getSpawnLocation() instanceof Spawnpoint) {
125  IContextSettings settings = player.getContextSettings();
126  if (settings != null && settings.isPlayerSpawnsReserved()) {
127 
128  getArena().getSpawns().reserve(player, (Spawnpoint) contextEvent.getSpawnLocation());
129 
130  }
131  }
132  }
133 
134  // display message from event
135  if (contextEvent.getMessage() != null) {
136  tell(contextEvent.getMessage());
137  }
138 
139  return true;
140  }

142  public final Result<Location> removePlayer(IArenaPlayer player, RemoveFromContextReason reason) {
143  PreCon.isValid(player instanceof ArenaPlayer);
144  PreCon.notNull(reason);
145 
146  // make sure the manager has the player
147  if (!_players.hasPlayer(player))
148  return new Result<>(false);
149 
150  if (reason != RemoveFromContextReason.CONTEXT_CHANGE) {
151 
152  // call pre remove event to see if the remove event is cancelled
153  if (_arena.getEventManager().call(this,
154  new PlayerPreRemoveFromContextEvent(_arena, player, this, reason))
155  .isCancelled())
156  return new Result<>(false);
157  }
158 
159  onPreRemovePlayer(player, reason);
160 
161  getArena().getSpawns().unreserve(player);
162 
163  // remove player from collection
164  _players.removePlayer(player, reason);
165 
166  Location restoreLocation = onRemovePlayer(player, reason);
167 
168  // call player removed event
169 
170  PlayerRemovedFromContextEvent removedEvent =
171  new PlayerRemovedFromContextEvent(_arena, player, this, getContext(), reason);
172  _arena.getEventManager().call(this, removedEvent);
173 
174  return new Result<>(true, restoreLocation);
175  }
