72  if (info == null)
73  throw new MissingExtensionAnnotation(extension);
74 
75  if (!TextUtils.isValidName(info.name(), 32)) {
76  throw new IllegalArgumentException(
77  "Arena Extension with an invalid name was detected: " + info.name());
78  }
79 
80  String key = info.name().toLowerCase();
81 
82  if (_extensionMap.containsKey(key)) {
83  Msg.warning("An Arena Extension was registered that overwrites another " +
84  "extension. Extension name: {0}.", info.name());
