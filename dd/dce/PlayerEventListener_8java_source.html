120  final ArenaPlayer player = ArenaPlayer.get(event.getPlayer());

122 
123  // respawn player in appropriate arena area.

155  @EventHandler(priority=EventPriority.LOWEST)

157 
158  IArenaPlayer player = ArenaPlayer.get(event.getPlayer());
159  IArena arena = player.getArena();
160 
161  if (arena == null)
162  return;
163 
164  Set<String> pvStarCommands = PVStarAPI.getPlugin().getDescription().getCommands().keySet();
165 
166  String[] comp = TextUtils.PATTERN_SPACE.split(event.getMessage());
167 
168  Matcher matcher = TextUtils.PATTERN_FILEPATH_SLASH.matcher(comp[0]);
169 
170  String command = matcher.replaceFirst("").toLowerCase();
171 
172  PlayerCommandEvent cmdEvent = new PlayerCommandEvent(
173  arena, player, player.getContextManager(), command, !pvStarCommands.contains(command));
174 
175  arena.getEventManager().call(this, cmdEvent);
176 
177  if (cmdEvent.isCancelled()) {
178  event.setMessage("/");
179  event.setCancelled(true);
